Javascript код для установки на сайт:

если на странице не подключен JQuery - данный код следует поместить в конце страницы перед закрывающим тегом </body>

<script>
var setup = {};

!function(b, o, a, t, v, i, e, w) {
  w = function () { v.version = "2.1", b.onw3d = b.onw3d || new Onw3dViewer(v), t && t(); },
  i = o.getElementsByTagName("head")[0], e = o.createElement("script"), e.type = "text/javascript",
  e.readyState ? e.onreadystatechange = function() {
    ("loaded" == e.readyState || "complete" == e.readyState) && (e.onreadystatechange = null, w());
  } : e.onload = function() { w(); }, e.src = a + "?v=" + v.version; i.appendChild(e);
} (window, document, "http://git.on-web.com.ua/js/onw3d.js", onReady, setup);
</script>

В случае с JQuery подключаем так:

$(function() {
  !function(b, o, a, t, v, i, e, w) {
  ...
});

onReady - это сallback функция. Вызовется, когда приложение будет готово к использованию, произойдет запуск приложения и загрузка указанной модели: (список доступных кодов моделей см. в таблице)

function onReady () {
    /* 1-й вариант: запуск в <div> контейнере */
    onw3d.init('код модели');
    /* или 2-й вариант: во всплывающем окне */
    onw3d.modal('код модели');
}

либо указать действия для кнопок:

function onReady () {
    document.getElementById('myElement').onclick = function() {
      onw3d.init('код модели');
    }  
}

setup - объект для доп. настроек

По умолчанию плеер ищет блок с идентификатором "scene", поэтому добавьте в нужном месте на странице <div id="scene"></div>, либо укажите в setup идентификатор своего блока - { container_id: "your_div_id" }

Помимо этого здесь можно переопределить следующие параметры:

  • player_path - путь к swf файлу плеера
  • models_path - к папке с файлам моделей
  • textures_path - к изображениям (текстурам)
  • xml_path - к файлам данных в формате XML

*Все пути (url) должны быть указаны абсолютными, например: { xml_path: 'http://domain.com/path_to_xml/' }

Это может понадобиться лишь в случае, если захотите разместить файлы на своем сервере. Скачать можно здесь.
